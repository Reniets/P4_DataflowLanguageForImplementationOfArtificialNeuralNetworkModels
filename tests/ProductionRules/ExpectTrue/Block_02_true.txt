block block_02 {
	channel:in input;
	channel:out output;
	
	blueprint() {
		this.input -> build Test(draw Singleton, [10, 10]) -> this.output;
	}

block Test {
    channel:in in;
    channel:out out;

    blueprint(blueprint blue, size s) {
        block b = build blue(s);

        this.in -> b -> this.out;
    }
}

block Singleton {
    channel:in in;
    channel:out out;

    blueprint(size s) {
        this.in -> this.out;
    }
}
