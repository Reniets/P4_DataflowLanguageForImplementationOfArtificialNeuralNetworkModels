// Test of channel access groups.
block chain_12 {
	mychannel:in input;
	mychannel:out output;
	
	blueprint() {
		block a = build Test();
		block b = build Test();
		block c = build Test();

		this.input -> a;
		(a.out, b.out, c.out) -> build LargeOpe() -> this.output;
	}
}

block Test {
    mychannel:in in;
    mychannel:out out;

    blueprint() {
        this.in -> build _Sigmoid() -> this.out;
    }
}

block LargeOpe {
    mychannel:in input1;
    mychannel:in input2;
    mychannel:in input3;
    mychannel:out output;

    blueprint() {
        operation mult1 = build Multiplication();
        operation mult2 = build Multiplication();
        operation mult3 = build _Multiplication();

        (this.input1, this.input1) -> mult1;
        (this.input2, this.input3) -> mult2;

        (mult1, mult2) -> mult3 -> this.output;
    }
}
